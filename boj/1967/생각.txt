  p,c,g=list(map(int,input().split()))
  adj[p].append((c,g))
  adj[c].append((p,g))

max_gram=-1e9
gram=0
lst=[]
idx=0
visited=[0]*(N+1)
def dfs(n,gram):
  global max_gram,idx
  visited[n]=1
  if(len(adj[n])==1):
    if(gram>max_gram):
      max_gram=gram
      idx=n

  for val in adj[n]:
    if(visited[val[0]]):continue
    dfs(val[0],gram+val[1])

dfs(1,0)
visited=[0]*(N+1)
max_gram=0
dfs(idx,0)
print(max_gram)
